window.onload = () => { const select = document.getElementById('select-jugador'); select.disabled = true; // Bloqueamos mientras carga select.innerHTML = ''; showLoading("Cargando lista de jugadores..."); google.script.run .withSuccessHandler(list => { hideLoading(); select.disabled = false; // Desbloqueamos if (list.length > 0 && list[0].startsWith("Error:")) { alert(list[0]); select.innerHTML = ''; return; } if (list.length === 0) { select.innerHTML = ''; return; } select.innerHTML = ''; list.forEach(name => { const option = document.createElement('option'); option.value = name; option.text = name; select.appendChild(option); }); }) .withFailureHandler(err => { hideLoading(); select.disabled = false; console.error(err); alert("Error al conectar con Google Sheets: " + err.message); select.innerHTML = ''; }) .getJugadores(); }; function addComensal(tipo, isFixed = false, datos = null) { if (tipo === 'Jugador') return; const container = document.getElementById('lista-comensales'); const id = Date.now(); const div = document.createElement('div'); div.className = 'comensal-card'; div.id = `c-${id}`; div.innerHTML = `

${tipo}

`; container.appendChild(div); } function cargarDatosPrevios() { const nombre = document.getElementById('select-jugador').value; if (!nombre) return; google.script.run.withSuccessHandler(res => { document.getElementById('lista-comensales').innerHTML = ''; if (res) { document.getElementById('observaciones').value = res.observaciones; res.comensales.forEach(c => { if (c.tipo === 'Acompañante') { addComensal(c.tipo, false, c); } }); } }).getDatosPrevios(nombre); } function enviarFormulario() { const btn = document.getElementById('btn-enviar'); const nombreJugador = document.getElementById('select-jugador').value; if (!nombreJugador) { alert("Por favor, selecciona tu nombre."); return; } btn.disabled = true; btn.innerText = "Enviando..."; const datos = { jugador: nombreJugador, observaciones: document.getElementById('observaciones').value, comensales: [] }; datos.comensales.push({ tipo: 'Jugador', nombre: nombreJugador, cenaV: 'N/A', comidaS1: 'N/A', comidaS2: 'N/A', comidaS3: 'N/A', cenaS: 'N/A', picnicD: 'N/A' }); document.querySelectorAll('.comensal-card').forEach(card => { datos.comensales.push({ tipo: card.querySelector('.c-tipo').value, nombre: card.querySelector('.c-nombre').value, cenaV: card.querySelector('.c-cenaV').value, comidaS1: card.querySelector('.c-comidaS1').value, comidaS2: card.querySelector('.c-comidaS2').value, comidaS3: card.querySelector('.c-comidaS3').value, cenaS: card.querySelector('.c-cenaS').value, picnicD: card.querySelector('.c-picnicD').value }); }); google.script.run.withSuccessHandler(msg => { alert(msg); btn.disabled = false; btn.innerText = "Enviar Elecciones"; }).guardarRespuestas(datos); }
